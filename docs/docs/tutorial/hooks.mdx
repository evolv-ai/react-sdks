## `useVariable()`

```tsx title="src/components/Button.tsx" showLineNumbers
import { useVariable } from '@evolv/nextjs';
import React, { FC } from 'react';

const Button: FC = (props) => {
	const buttonText = useVariable('', 'Failover Value');
	
	return (
		<button>
			{buttonText}
		</button>
	);
};
```
